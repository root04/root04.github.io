When using find/find_by, returning result will be ...


foo.find(1)                                          # object of ID == 1
foo.find([1,2,3])                                    # Array of object ID == 1,2,3
foo.find(<ID of no matching record>)       # ActiveRecord::RecordNotFound
foo.find_by(<ID of none existing record>) # nil
